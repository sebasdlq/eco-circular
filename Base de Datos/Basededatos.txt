-- ======================================================
-- ECOCIRCULAR AI - SCHEMA PRINCIPAL (POSTGRESQL)
-- SIN POSTGIS, solo lat/lon numéricos
-- ======================================================

-- Habilitar extensión UUID (si no existe)
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- ======================================================
-- 1. ENUMs (dominios cerrados)
-- ======================================================

CREATE TYPE rol_usuario AS ENUM (
    'CIUDADANO',
    'ESTUDIANTE',
    'OPERADOR',
    'DOCENTE',
    'MUNICIPALIDAD',
    'RECICLADORA',
    'AUDITOR'
);

CREATE TYPE estado_entrega AS ENUM (
    'BORRADOR',
    'VALIDADA',
    'CONTABILIZADA',
    'AJUSTADA'
);

CREATE TYPE estado_lote AS ENUM (
    'ABIERTO',
    'CERRADO',
    'DESPACHADO',
    'RECIBIDO'
);

CREATE TYPE estado_mision AS ENUM (
    'CREADA',
    'ASIGNADA',
    'EN_PROGRESO',
    'COMPLETADA',
    'EXPIRADA'
);

CREATE TYPE estado_campania AS ENUM (
    'BORRADOR',
    'PROGRAMADA',
    'PUBLICADA',
    'CERRADA',
    'EVALUADA'
);

CREATE TYPE tipo_ranking AS ENUM (
    'USUARIO',
    'CURSO',
    'ZONA'
);

CREATE TYPE owner_tipo AS ENUM (
    'CURSO',
    'MUNICIPALIDAD',
    'ESCUELA'
);

CREATE TYPE unidad_medida AS ENUM (
    'KG',
    'UNIDAD',
    'LITRO'
);

-- ======================================================
-- 2. TABLAS PRINCIPALES
-- ======================================================

-- Tenant (municipio, red de escuelas, alianza)
CREATE TABLE tenant (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    nombre VARCHAR(100) NOT NULL,
    config JSONB NOT NULL DEFAULT '{}',
    estado VARCHAR(20) NOT NULL DEFAULT 'ACTIVO',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Usuario (persona física)
CREATE TABLE usuario (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID NOT NULL REFERENCES tenant(id) ON DELETE CASCADE,
    rol rol_usuario NOT NULL,
    nombre VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE,
    telefono VARCHAR(20),
    escuela_id UUID NULL,
    curso_id UUID NULL,
    punto_verde_id UUID NULL,
    activo BOOLEAN DEFAULT true,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    CONSTRAINT check_estudiante_escuela CHECK (
        (rol IN ('ESTUDIANTE', 'DOCENTE') AND escuela_id IS NOT NULL) OR
        (rol NOT IN ('ESTUDIANTE', 'DOCENTE'))
    ),
    CONSTRAINT check_operador_punto CHECK (
        (rol = 'OPERADOR' AND punto_verde_id IS NOT NULL) OR
        (rol != 'OPERADOR')
    )
);

-- Escuela
CREATE TABLE escuela (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID NOT NULL REFERENCES tenant(id) ON DELETE CASCADE,
    nombre VARCHAR(150) NOT NULL,
    direccion TEXT,
    telefono VARCHAR(20),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Curso
CREATE TABLE curso (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    escuela_id UUID NOT NULL REFERENCES escuela(id) ON DELETE CASCADE,
    nombre VARCHAR(50) NOT NULL,
    grado INT,
    paralelo VARCHAR(10),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- FKs de usuario (después de crear escuela y curso)
ALTER TABLE usuario
    ADD CONSTRAINT fk_usuario_escuela FOREIGN KEY (escuela_id) REFERENCES escuela(id) ON DELETE SET NULL,
    ADD CONSTRAINT fk_usuario_curso FOREIGN KEY (curso_id) REFERENCES curso(id) ON DELETE SET NULL;

-- Punto Verde (sin PostGIS)
CREATE TABLE punto_verde (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID NOT NULL REFERENCES tenant(id) ON DELETE CASCADE,
    nombre VARCHAR(100) NOT NULL,
    latitud DOUBLE PRECISION,
    longitud DOUBLE PRECISION,
    horarios JSONB,
    capacidad INT,
    estado VARCHAR(20) DEFAULT 'ACTIVO',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- FK de usuario a punto_verde (operador)
ALTER TABLE usuario
    ADD CONSTRAINT fk_usuario_punto_verde FOREIGN KEY (punto_verde_id) REFERENCES punto_verde(id) ON DELETE SET NULL;

-- Material (global)
CREATE TABLE material (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    nombre VARCHAR(50) NOT NULL UNIQUE,
    categoria VARCHAR(50),
    puntos_por_kg DECIMAL(10,2) NOT NULL,
    co2_por_kg DECIMAL(10,3) NOT NULL,
    activo BOOLEAN DEFAULT true,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Entrega (cabecera)
CREATE TABLE entrega (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID NOT NULL REFERENCES tenant(id) ON DELETE CASCADE,
    usuario_id UUID NOT NULL REFERENCES usuario(id),
    punto_verde_id UUID NOT NULL REFERENCES punto_verde(id),
    estado estado_entrega NOT NULL DEFAULT 'BORRADOR',
    fecha TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    version INT DEFAULT 1,
    validado_por UUID REFERENCES usuario(id),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Detalle de entrega
CREATE TABLE entrega_detalle (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    entrega_id UUID NOT NULL REFERENCES entrega(id) ON DELETE CASCADE,
    material_id UUID NOT NULL REFERENCES material(id),
    cantidad DECIMAL(10,2) NOT NULL CHECK (cantidad > 0),
    unidad unidad_medida DEFAULT 'KG',
    puntos DECIMAL(10,2) NOT NULL,
    co2_estimado DECIMAL(10,3) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Lote
CREATE TABLE lote (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    punto_verde_id UUID NOT NULL REFERENCES punto_verde(id),
    estado estado_lote NOT NULL DEFAULT 'ABIERTO',
    fecha_apertura TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    fecha_cierre TIMESTAMP WITH TIME ZONE,
    fecha_despacho TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Lote-Entrega
CREATE TABLE lote_entrega (
    lote_id UUID NOT NULL REFERENCES lote(id) ON DELETE CASCADE,
    entrega_id UUID NOT NULL REFERENCES entrega(id) ON DELETE CASCADE,
    PRIMARY KEY (lote_id, entrega_id)
);

-- Misión
CREATE TABLE mision (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID NOT NULL REFERENCES tenant(id) ON DELETE CASCADE,
    owner_tipo owner_tipo NOT NULL,
    owner_id UUID NOT NULL,
    nombre VARCHAR(100) NOT NULL,
    reglas JSONB NOT NULL,
    fecha_inicio TIMESTAMP WITH TIME ZONE NOT NULL,
    fecha_fin TIMESTAMP WITH TIME ZONE NOT NULL,
    recompensa JSONB,
    estado estado_mision DEFAULT 'CREADA',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Progreso de misión por usuario
CREATE TABLE mision_usuario (
    mision_id UUID NOT NULL REFERENCES mision(id) ON DELETE CASCADE,
    usuario_id UUID NOT NULL REFERENCES usuario(id) ON DELETE CASCADE,
    progreso JSONB DEFAULT '{}',
    completada BOOLEAN DEFAULT false,
    fecha_completada TIMESTAMP WITH TIME ZONE,
    PRIMARY KEY (mision_id, usuario_id)
);

-- Badge
CREATE TABLE badge (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    nombre VARCHAR(50) NOT NULL,
    descripcion TEXT,
    condiciones JSONB,
    imagen_url TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Usuario-Badge
CREATE TABLE usuario_badge (
    usuario_id UUID NOT NULL REFERENCES usuario(id) ON DELETE CASCADE,
    badge_id UUID NOT NULL REFERENCES badge(id) ON DELETE CASCADE,
    fecha_obtencion TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    PRIMARY KEY (usuario_id, badge_id)
);

-- Ranking (snapshots)
CREATE TABLE ranking (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID NOT NULL REFERENCES tenant(id),
    tipo tipo_ranking NOT NULL,
    referencial_id VARCHAR(100) NOT NULL,
    periodo VARCHAR(20) NOT NULL,
    score DECIMAL(10,2) NOT NULL,
    snapshot_ts TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Campaña
CREATE TABLE campania (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID NOT NULL REFERENCES tenant(id),
    nombre VARCHAR(100) NOT NULL,
    estado estado_campania DEFAULT 'BORRADOR',
    fecha_inicio TIMESTAMP WITH TIME ZONE NOT NULL,
    fecha_fin TIMESTAMP WITH TIME ZONE NOT NULL,
    segmentacion JSONB,
    mensajes JSONB,
    resultados JSONB,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Auditoría
CREATE TABLE evento_auditoria (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tenant_id UUID NOT NULL REFERENCES tenant(id),
    accion VARCHAR(100) NOT NULL,
    actor_id UUID NOT NULL,
    actor_rol rol_usuario NOT NULL,
    recurso_tipo VARCHAR(50) NOT NULL,
    recurso_id UUID NOT NULL,
    payload JSONB NOT NULL,
    razon TEXT,
    timestamp TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- ======================================================
-- 3. ÍNDICES ESTRATÉGICOS (sin PostGIS)
-- ======================================================

CREATE INDEX idx_entrega_tenant_fecha ON entrega(tenant_id, fecha);
CREATE INDEX idx_entrega_usuario ON entrega(usuario_id);
CREATE INDEX idx_entrega_punto_verde ON entrega(punto_verde_id);
CREATE INDEX idx_entrega_estado ON entrega(estado);

CREATE INDEX idx_lote_estado ON lote(estado);
CREATE INDEX idx_lote_punto_verde ON lote(punto_verde_id);

CREATE INDEX idx_mision_tenant_periodo ON mision(tenant_id, fecha_inicio, fecha_fin);
CREATE INDEX idx_mision_owner ON mision(owner_tipo, owner_id);

CREATE INDEX idx_ranking_tenant_tipo_periodo ON ranking(tenant_id, tipo, periodo);
CREATE INDEX idx_ranking_referencial ON ranking(referencial_id);

CREATE INDEX idx_usuario_tenant_rol ON usuario(tenant_id, rol);
CREATE INDEX idx_usuario_email ON usuario(email);

CREATE INDEX idx_punto_verde_tenant ON punto_verde(tenant_id);
-- No se necesita índice espacial sin PostGIS

CREATE INDEX idx_auditoria_tenant_timestamp ON evento_auditoria(tenant_id, timestamp);
CREATE INDEX idx_auditoria_recurso ON evento_auditoria(recurso_tipo, recurso_id);
